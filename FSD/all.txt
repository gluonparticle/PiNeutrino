<!DOCTYPE html>
<html>
<head>
<title>String Manipulation</title>
<script>
// Function to check palindrome (method 1)
function isPalindrome1(str) {
    for (let i = 0; i < str.length / 2; i++) {
        if (str.charAt(i) != str.charAt(str.length - 1 - i)) {
            return false;
        }
    }
    return true;
}

// Function to check palindrome (method 2)
function isPalindrome(str) {
    let cleanedStr = str.replace(/[^A-Za-z0-9]/g, '').toLowerCase();
    return cleanedStr === cleanedStr.split('').reverse().join('');
}

// Run after page loads
function runProgram() {
    let userInput = prompt("Enter a string:");

    let lengthOfString = userInput.length;

    // Extract 'JavaScript'
    let extractedWord = userInput.includes("JavaScript")
        ? userInput.substring(
            userInput.indexOf("JavaScript"),
            userInput.indexOf("JavaScript") + "JavaScript".length
        )
        : "JavaScript not found";

    // Replace word
    let newString = userInput.replace("JavaScript", "TypeScript");

    // Palindrome check
    let palindromeCheck = isPalindrome(userInput);

    console.log("New String after Replacement:", newString);

    alert(
        "Length: " + lengthOfString +
        "\nExtracted Word: " + extractedWord +
        "\nNew String: " + newString +
        "\nPalindrome: " + palindromeCheck
    );
}
</script>
</head>

<body onload="runProgram()">
</body>
</html>




3.1 html 
<!DOCTYPE html>
<html>
<head>
    <title>One Student</title>

    <script>
        function createStudent() {
            let student = {
                name: prompt("Enter the student's name:"),
                grade: parseInt(prompt("Enter the student's grade:")),
                subjects: prompt("Enter subjects (comma separated):")
                    .split(',')
                    .map(s => s.trim()),
                passed: function() {
                    return this.grade >= 50;
                }
            };
            return student;
        }

        function displayInfo(student) {
            for (let key in student) {
                if (typeof student[key] !== 'function') {
                    console.log(key + ": " + student[key]);
                }
            }

            // ✅ Correctly calling the function
            console.log("Passed: " + (student.passed() ? "Yes" : "No"));
        }

        let student = createStudent();
        displayInfo(student);
    </script>
</head>

<!DOCTYPE html>
<html>
<head>        
    <title>Multiple Students</title>

    <script>
        function createStudent() {
            let student = {
                name: prompt("Enter student name:"),
                grade: parseFloat(prompt("Enter student grade:")),
                subjects: prompt("Enter student subjects (comma separated):").split(',').map(s => s.trim()),
                passed: function() {
                    return this.grade >= 50;
                }
            };
            return student;
        }

        function displayInfo(student) {
            for (let key in student) {
                if (typeof student[key] !== 'function') {
                    console.log(key + ": " + student[key]);
                }
            }
            console.log("Passed: " + (student.passed() ? "Yes" : "No"));
            console.log("------------");
        }

        let numStudents = parseInt(prompt("How many students?"));

        for (let i = 0; i < numStudents; i++) {
            console.log("Enter details for student " + (i + 1));
            let student = createStudent();
            displayInfo(student);
        }
    </script>
</head>

<body>
</body>

</html>
